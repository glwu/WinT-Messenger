185 void PeerManager::readBroadcastDatagram() {
186  while (broadcastSocket.hasPendingDatagrams()) {
187  QHostAddress senderIp;
188  quint16 senderPort;
189  QByteArray datagram;
190  datagram.resize(broadcastSocket.pendingDatagramSize());
191  if (broadcastSocket.readDatagram(datagram.data(), datagram.size(),
192  &senderIp, &senderPort) == -1)
193  continue;
194 
195  QList<QByteArray> list = datagram.split('@');
196 
197  if (list.size() != 3)
198  continue;
199 
200  int senderFilePort = list.at(1).toInt();
201  int senderMessagePort = list.at(2).toInt();
202 
203  if (isLocalHostAddress(senderIp) && senderMessagePort == m_serverPort && senderFilePort == f_serverPort)
204  continue;
205 
206  if (!client->hasConnection(senderIp)) {
207  FConnection *fc = new FConnection(this);
208  MConnection *mc = new MConnection(this);
209 
210  fc->connectToHost(senderIp, senderFilePort);
211  mc->connectToHost(senderIp, senderMessagePort);
212 
213  emit newFileConnection(fc);
214  emit newMessageConnection(mc);
215  }
216  }
217 }

223 void PeerManager::updateAddresses() {
224  broadcastAddresses.clear();
225  ipAddresses.clear();
226  foreach (QNetworkInterface interface, QNetworkInterface::allInterfaces()) {
227  foreach (QNetworkAddressEntry entry, interface.addressEntries()) {
228  QHostAddress broadcastAddress = entry.broadcast();
229  if (broadcastAddress != QHostAddress::Null && entry.ip() != QHostAddress::LocalHost) {
230  broadcastAddresses << broadcastAddress;

232  }
233  }
