141 bool Client::hasConnection(const QHostAddress &senderIp, int senderPort) const {
142  if (senderPort == -1)
143  return message_peers.contains(senderIp);
144 
145  if (!message_peers.contains(senderIp))
146  return false;
147 
148  QList<MConnection *> connections = message_peers.values(senderIp);
149  foreach (MConnection *connection, connections) {
150  if (connection->peerPort() == senderPort)
151  return true;
152  }
153 
154  return false;
155 }
