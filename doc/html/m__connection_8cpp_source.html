143 void MConnection::processReadyRead() {
144  if (state == WaitingForGreeting) {
145  if (!readProtocolHeader())
146  return;
147  if (currentDataType != Greeting) {
148  abort();
149  return;
150  }
151 
152  state = ReadingGreeting;
153  }
154 
155  if (state == ReadingGreeting) {
156  if (!hasEnoughData())
157  return;
158 
159  buffer = read(numBytesForCurrentDataType);
160  if (buffer.size() != numBytesForCurrentDataType) {
161  abort();
162  return;
163  }
164 
165  QList<QByteArray> list = buffer.split('@');
166 
167  username = QString::fromUtf8(list.at(0)) + '@' +
168  peerAddress().toString();
169  if (list.count() > 1)
170  userface = QString(list.at(1));
171  else
172  userface = "/system/generic-user.png";
173 
174  currentDataType = Undefined;
175  numBytesForCurrentDataType = 0;
176  buffer.clear();
177 
178  if (!isValid()) {
179  abort();
180  return;
181  }
182 
183  if (!isGreetingMessageSent)
184  sendGreetingMessage();
185 
186  pingTimer.start();
187  pongTime.start();
188  state = ReadyForUse;
189  emit readyForUse();
190  }
191 
192  while (bytesAvailable() > 0) {
193  if (currentDataType == Undefined)
194  if (!readProtocolHeader())
195  return;
196 
197  if (!hasEnoughData())
198  return;
199 
200  processData();
201  }
